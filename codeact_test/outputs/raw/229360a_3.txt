def get_all_pages(api_func, **kwargs):
    results = []
    page_index = 0
    while True:
        response = api_func(access_token=access_token, page_index=page_index, page_limit=20, **kwargs)
        if not response:
            break
        results.extend(response)
        if len(response) < 20:
            break
        page_index += 1
    return results
