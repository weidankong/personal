Code:
filtered_songs = []
for rec in all_recommendations:
    song_details = apis.spotify.show_song(song_id=rec['song_id'])
    if 'release_date' in song_details and 'genre' in song_details:
        release_year = int(song_details['release_date'][:4])
        genre = song_details['genre'].lower()
        if release_year in [2022, 2023] and 'r&b' in genre:
            filtered_songs.append(song_details)
