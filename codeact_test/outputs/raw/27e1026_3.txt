access_token = access_token_resp['access_token']

def get_all_pages(api_func, **kwargs):
    items = []
    page_index = 0
    while True:
        resp = api_func(access_token=access_token, page_index=page_index, page_limit=20, **kwargs)
        if not resp:
            break
        items.extend(resp)
        if len(resp) < 20:
            break
        page_index += 1
    return items
