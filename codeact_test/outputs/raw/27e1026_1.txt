Code:
access_token = login_response['access_token']
song_ids = set()

# Fetch song library
page = 0
while True:
    res = apis.spotify.show_song_library(access_token=access_token, page_index=page, page_limit=20)
    if not res:
        break
    for s in res:
        song_ids.add(s['song_id'])
    if len(res) < 20:
        break
    page += 1

# Fetch album library
page = 0
while True:
    res = apis.spotify.show_album_library(access_token=access_token, page_index=page, page_limit=20)
    if not res:
        break
    for a in res:
        for sid in a.get('song_ids', []):
            song_ids.add(sid)
    if len(res) < 20:
        break
    page += 1

# Fetch playlist library
page = 0
while True:
    res = apis.spotify.show_playlist_library(access_token=access_token, page_index=page, page_limit=20)
    if not res:
        break
    for p in res:
        for sid in p.get('song_ids', []):
            song_ids.add(sid)
    if len(res) < 20:
        break
    page += 1

print(f"Total unique song IDs: {len(song_ids)}")

oldest_song_title = None
oldest_release_date = None

for sid in song_ids:
    song_details = apis.spotify.show_song(song_id=sid)
    if 'release_date' in song_details:
        release_date = song_details['release_date']
        if oldest_release_date is None or release_date < oldest_release_date:
            oldest_release_date = release_date
            oldest_song_title = song_details['title']

print(f"Oldest song: {oldest_song_title}, released on: {oldest_release_date}")
