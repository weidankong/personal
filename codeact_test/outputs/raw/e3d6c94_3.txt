Code:
# Get details for all recommended songs and filter by R&B genre and 2023 release year
rb_songs_2023 = []
for rec in all_recs:
    details = apis.spotify.show_song(song_id=rec['song_id'])
    if details.get('genre') == 'R&B' and details.get('release_date', '').startswith('2023'):
