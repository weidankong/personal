access_token = login_response['access_token']
all_recommendations = []
page_index = 0
while True:
    recs = apis.spotify.show_recommendations(access_token=access_token, page_index=page_index, page_limit=20)
    if isinstance(recs, dict):
        print("Error:", recs)
        break
    if not recs:
        break
    all_recommendations.extend(recs)
    if len(recs) < 20:
        break
    page_index += 1

from collections import Counter
artist_counts = Counter()
for song in all_recommendations:
    for artist in song.get('artists', []):
        artist_counts[artist['name']] += 1

print("Total songs:", len(all_recommendations))
print("Top artists:", artist_counts.most_common(5))
