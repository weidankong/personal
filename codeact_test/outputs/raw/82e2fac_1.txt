access_token = login_response['access_token']
all_song_ids = set()
page_index = 0
while True:
    playlists_page = apis.spotify.show_playlist_library(access_token=access_token, page_limit=20, page_index=page_index)
    if not playlists_page or 'message' in playlists_page:
        break
    for p in playlists_page:
        all_song_ids.update(p.get('song_ids', []))
    if len(playlists_page) < 20:
        break
    page_index += 1

print(f"Total unique songs in playlists: {len(all_song_ids)}")
