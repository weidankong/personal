# Let's write a script to iterate through all songs in the album library, check if they are liked, and then either create or update the review to 4 stars.
access_token = login_response['access_token']
user_email = 'jes.mill@gmail.com'

albums = []
page_index = 0
while True:
    page = apis.spotify.show_album_library(access_token=access_token, page_index=page_index, page_limit=20)
    if not page:
        break
    albums.extend(page)
    if len(page) < 20:
        break
    page_index += 1

all_song_ids = set()
for album in albums:
    all_song_ids.update(album['song_ids'])

updated_count = 0
created_count = 0
skipped_count = 0

for song_id in all_song_ids:
    privates = apis.spotify.show_song_privates(song_id=song_id, access_token=access_token)
    if privates.get('liked'):
        reviews = apis.spotify.show_song_reviews(song_id=song_id, user_email=user_email, page_limit=20)
        my_review = None
        for r in reviews:
            if r['user']['email'] == user_email:
                my_review = r
                break
        
        if my_review:
            if my_review['rating'] < 4:
                apis.spotify.update_song_review(review_id=my_review['song_review_id'], access_token=access_token, rating=4)
                updated_count += 1
            else:
                skipped_count += 1
        else:
            apis.spotify.review_song(song_id=song_id, rating=4, access_token=access_token)
            created_count += 1

print(f"Updated: {updated_count}, Created: {created_count}, Skipped: {skipped_count}")
Output: Updated: 3, Created: 17, Skipped: 0
