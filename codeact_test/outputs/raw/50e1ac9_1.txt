Code:
rb_songs = []
for sid in all_song_ids:
    song_details = apis.spotify.show_song(song_id=sid)
    if "genre" in song_details and song_details["genre"].lower() == "r&b":
        rb_songs.append({
            "title": song_details["title"],
            "play_count": song_details.get("play_count", 0)
        })

rb_songs.sort(key=lambda x: x["play_count"], reverse=True)
top_4 = rb_songs[:4]
titles = [s["title"] for s in top_4]
