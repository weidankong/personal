access_token = login_response['access_token']

def get_all(api_func, **kwargs):
    items = []
    page = 0
    while True:
        res = api_func(access_token=access_token, page_index=page, page_limit=20, **kwargs)
        if not res:
            break
        items.extend(res)
        if len(res) < 20:
            break
        page += 1
    return items

songs = get_all(apis.spotify.show_song_library)

from datetime import datetime

newest_song = None
newest_date = None

for sid in song_ids:
    song_details = apis.spotify.show_song(song_id=sid)
    if 'release_date' in song_details:
        release_date_str = song_details['release_date']
        try:
            release_date = datetime.fromisoformat(release_date_str)
        except ValueError:
            release_date = datetime.strptime(release_date_str, "%Y-%m-%dT%H:%M:%S")
            
        if newest_date is None or release_date > newest_date:
            newest_date = release_date
            newest_song = song_details

print(f"Newest song: {newest_song['title']}, Released at: {newest_date}")
