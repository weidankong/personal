def escape_csv(field):
    s = str(field)
    if ',' in s or '"' in s or '\n' in s:
        return '"' + s.replace('"', '""') + '"'
    return s

lines = ["Title,Artists"]
for song_id, song_data in all_songs.items():
    title = escape_csv(song_data['title'])
    artists = escape_csv(song_data['artists'])
