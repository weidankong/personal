def get_all_pages(api_func, access_token, **kwargs):
    results = []
    page_index = 0
    while True:
        res = api_func(access_token=access_token, page_index=page_index, page_limit=20, **kwargs)
        if not res:
            break
        results.extend(res)
        if len(res) < 20:
            break
        page_index += 1
    return results
